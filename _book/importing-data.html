this is useful because we can fix the column types like this.

my_iris <- read_csv(
    file = "iris.csv",
    col_types = cols(
      Sepal.Length = col_double(),
      Sepal.Width = col_double(),
      Petal.Length = col_double(),
      Petal.Width = col_double(),
      Species = col_character()
    )
)
